01
Crear la base de datos
CREATE DATABASE · USE · SHOW · DROP
▼

Antes de crear cualquier tabla necesitamos un contenedor: la base de datos. En MySQL esto se hace con CREATE DATABASE.

💡
Usaremos un caso de gestión de pedidos simple: clientes que realizan pedidos con productos. Tres entidades, relaciones claras.

Comandos esenciales

SQL
-- 1. Verificar bases de datos existentes
SHOW DATABASES;

-- 2. Crear la base de datos con encoding correcto
CREATE DATABASE tienda_db
  CHARACTER SET utf8mb4
  COLLATE utf8mb4_unicode_ci;

-- 3. Seleccionarla para trabajar
USE tienda_db;

-- 4. Ver en qué base de datos estamos
SELECT DATABASE();

-- 5. Eliminar (cuidado: borra TODO)
-- DROP DATABASE tienda_db;
⚠️
utf8mb4 es el encoding correcto en MySQL moderno. Soporta emojis, tildes y caracteres especiales del español. No usar solo utf8.
02
Crear las tablas
CREATE TABLE · tipos de datos · PRIMARY KEY · NOT NULL
▼

Creamos las tablas respetando el orden: primero las tablas independientes (sin FK) y luego las dependientes.

Modelo de datos — caso tienda

Tabla clientes (independiente, sin FK)

SQL
CREATE TABLE clientes (
  id_cliente      INT           NOT NULL AUTO_INCREMENT,
  nombre          VARCHAR(100)  NOT NULL,
  correo          VARCHAR(150)  NOT NULL UNIQUE,
  telefono        VARCHAR(20)   NULL,
  fecha_registro  DATETIME      NOT NULL DEFAULT NOW(),
  PRIMARY KEY (id_cliente)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

Tabla productos (independiente, sin FK)

SQL
CREATE TABLE productos (
  id_producto  INT             NOT NULL AUTO_INCREMENT,
  nombre       VARCHAR(100)    NOT NULL,
  precio       DECIMAL(10,2)   NOT NULL CHECK (precio > 0),
  stock        INT             NOT NULL DEFAULT 0,
  PRIMARY KEY (id_producto)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

03
Claves foráneas (FOREIGN KEY)
REFERENCES · integridad referencial · tabla dependiente
▼

La tabla pedidos referencia a clientes y productos. Debe crearse después de ellas.

📌
Regla de oro: la tabla que tiene la FK siempre se crea al final. La tabla referenciada debe existir primero.
SQL
CREATE TABLE pedidos (
  id_pedido    INT        NOT NULL AUTO_INCREMENT,
  id_cliente   INT        NOT NULL,
  id_producto  INT        NOT NULL,
  cantidad     INT        NOT NULL DEFAULT 1,
  fecha_pedido DATETIME   NOT NULL DEFAULT NOW(),
  estado       ENUM('pendiente','enviado','entregado','cancelado')
               NOT NULL DEFAULT 'pendiente',

  PRIMARY KEY (id_pedido),

  -- Definición de claves foráneas
  CONSTRAINT fk_pedido_cliente
    FOREIGN KEY (id_cliente) REFERENCES clientes(id_cliente),

  CONSTRAINT fk_pedido_producto
    FOREIGN KEY (id_producto) REFERENCES productos(id_producto)

) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

Anatomía de la FK

ParteQué hace
CONSTRAINT fk_nombreNombre identificador de la restricción (opcional pero recomendado)
FOREIGN KEY (col)Columna local que actúa como FK
REFERENCES tabla(col)Tabla y columna padre que se referencia
ON DELETE / ON UPDATEComportamiento cuando el padre se elimina o actualiza
⚠️
El motor ENGINE=InnoDB es obligatorio para que las FK funcionen en MySQL. MyISAM las ignora.
04
Índices (INDEX)
CREATE INDEX · rendimiento · UNIQUE INDEX
▼

Los índices aceleran las consultas WHERE, JOIN y ORDER BY. Las FK crean índices automáticamente, pero podemos agregar más.

SQL
-- Índice simple para búsquedas por nombre de cliente
CREATE INDEX idx_cliente_nombre
  ON clientes(nombre);

-- Índice único (garantiza no duplicados)
CREATE UNIQUE INDEX idx_cliente_correo
  ON clientes(correo);

-- Índice compuesto: filtra por estado y fecha juntos
CREATE INDEX idx_pedido_estado_fecha
  ON pedidos(estado, fecha_pedido);

-- Ver índices de una tabla
SHOW INDEX FROM pedidos;

-- Eliminar un índice
DROP INDEX idx_cliente_nombre ON clientes;
✅
Crea índices en columnas que usas frecuentemente en WHERE, JOIN ON y ORDER BY. Evita indexar columnas con pocos valores distintos (ej: un campo booleano).
05
Insertar datos (INSERT)
INSERT INTO · VALUES · INSERT múltiple · orden de inserción
▼

Al igual que con las tablas, el orden importa: primero insertamos en las tablas padre y luego en las hijas.

SQL
-- 1. Primero: clientes (tabla padre)
INSERT INTO clientes (nombre, correo, telefono) VALUES
  ('Ana Gómez',    'ana@correo.com',    '3001234567'),
  ('Luis Torres',   'luis@correo.com',   '3109876543'),
  ('María Ríos',    'maria@correo.com',  NULL);

-- 2. Segundo: productos (tabla padre)
INSERT INTO productos (nombre, precio, stock) VALUES
  ('Cuaderno',      4500.00,  50),
  ('Lapicero',      1200.00,  200),
  ('Mochila',       85000.00, 15);

-- 3. Tercero: pedidos (tabla hija, referencia a las anteriores)
INSERT INTO pedidos (id_cliente, id_producto, cantidad, estado) VALUES
  (1, 2, 3, 'pendiente'),   -- Ana pide 3 lapiceros
  (2, 1, 1, 'enviado'),     -- Luis pide 1 cuaderno
  (1, 3, 2, 'pendiente');   -- Ana pide 2 mochilas

-- Ver qué se insertó
SELECT * FROM clientes;
SELECT * FROM pedidos;
⚠️
Si intentas insertar en pedidos un id_cliente que no existe en clientes, MySQL lanzará error: Cannot add or update a child row: a foreign key constraint fails.
06
Modificar estructura (ALTER TABLE)
ADD COLUMN · MODIFY · DROP COLUMN · RENAME
▼

ALTER TABLE permite modificar la estructura de tablas ya creadas sin perder los datos existentes.

SQL
-- Agregar una columna nueva
ALTER TABLE clientes
  ADD COLUMN ciudad VARCHAR(80) NULL AFTER telefono;

-- Cambiar tipo o restricciones de una columna
ALTER TABLE clientes
  MODIFY COLUMN telefono VARCHAR(30) NULL;

-- Renombrar una columna (MySQL 8+)
ALTER TABLE clientes
  RENAME COLUMN ciudad TO municipio;

-- Eliminar una columna
ALTER TABLE clientes
  DROP COLUMN municipio;

-- Agregar una FK a tabla existente
ALTER TABLE pedidos
  ADD CONSTRAINT fk_nueva
  FOREIGN KEY (id_cliente) REFERENCES clientes(id_cliente);

-- Eliminar una FK existente
ALTER TABLE pedidos
  DROP FOREIGN KEY fk_pedido_cliente;

-- Ver la estructura actual de una tabla
DESCRIBE clientes;
SHOW CREATE TABLE pedidos;
✅
Antes de hacer DROP COLUMN en producción, siempre haz un SELECT de esa columna para confirmar que los datos ya no son necesarios.
07
Eliminación en cascada (ON DELETE CASCADE)
CASCADE · SET NULL · RESTRICT · NO ACTION · ON UPDATE
▼

Las opciones de cascada definen qué pasa con los registros hijos cuando el padre se elimina o actualiza. Es la clave de la integridad referencial automática.

Las 4 opciones disponibles

Opción¿Qué hace al eliminar/actualizar el padre?Cuándo usarla
CASCADEElimina/actualiza automáticamente los hijosRegistros dependientes sin sentido solos (ej: pedidos de un cliente)
SET NULLPone NULL en la FK del hijoLa FK del hijo debe admitir NULL; el registro hijo sigue existiendo
RESTRICTBloquea la eliminación si hay hijosCuando NO queremos borrar accidentalmente registros padre con hijos
NO ACTIONIgual que RESTRICT en MySQLComportamiento por defecto si no se especifica nada

Ejemplo: CASCADE en acción

SQL
-- Primero eliminamos la FK anterior
ALTER TABLE pedidos
  DROP FOREIGN KEY fk_pedido_cliente;

-- Redefinimos con CASCADE
ALTER TABLE pedidos
  ADD CONSTRAINT fk_pedido_cliente
  FOREIGN KEY (id_cliente)
  REFERENCES clientes(id_cliente)
  ON DELETE CASCADE
  ON UPDATE CASCADE;

-- Prueba: al borrar a Ana (id=1), sus pedidos se borran solos
DELETE FROM clientes WHERE id_cliente = 1;

-- Verificar: ya no existen pedidos de Ana
SELECT * FROM pedidos WHERE id_cliente = 1;

Ejemplo: SET NULL

SQL
-- La columna id_producto debe admitir NULL primero
ALTER TABLE pedidos
  MODIFY COLUMN id_producto INT NULL;

-- Redefinir FK con SET NULL
ALTER TABLE pedidos
  DROP FOREIGN KEY fk_pedido_producto;

ALTER TABLE pedidos
  ADD CONSTRAINT fk_pedido_producto
  FOREIGN KEY (id_producto)
  REFERENCES productos(id_producto)
  ON DELETE SET NULL
  ON UPDATE CASCADE;

-- Al borrar un producto, los pedidos quedan con id_producto = NULL
DELETE FROM productos WHERE id_producto = 3;
⚠️
ON DELETE CASCADE es poderoso pero peligroso: un solo DELETE puede eliminar cientos de registros en cadena. Úsalo solo cuando los hijos no tienen sentido sin el padre.
08
Script completo — listo para ejecutar
Script final · orden correcto · DROP IF EXISTS · datos de prueba
▼

Este es el script completo y ordenado: puedes ejecutarlo desde cero en cualquier momento. Incluye DROP IF EXISTS para poder relanzarlo sin errores.

SQL — Script completo tienda_db
-- ==========================================
-- SCRIPT COMPLETO: tienda_db
-- MySQL / MariaDB · ENGINE InnoDB
-- ==========================================

-- 1. Crear y seleccionar la BD
DROP DATABASE IF EXISTS tienda_db;
CREATE DATABASE tienda_db
  CHARACTER SET utf8mb4
  COLLATE utf8mb4_unicode_ci;
USE tienda_db;

-- ==========================================
-- 2. TABLAS PADRE (sin FK)
-- ==========================================

CREATE TABLE clientes (
  id_cliente      INT           NOT NULL AUTO_INCREMENT,
  nombre          VARCHAR(100)  NOT NULL,
  correo          VARCHAR(150)  NOT NULL,
  telefono        VARCHAR(20)   NULL,
  fecha_registro  DATETIME      NOT NULL DEFAULT NOW(),
  PRIMARY KEY (id_cliente),
  UNIQUE (correo)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE productos (
  id_producto  INT            NOT NULL AUTO_INCREMENT,
  nombre       VARCHAR(100)   NOT NULL,
  precio       DECIMAL(10,2)  NOT NULL,
  stock        INT            NOT NULL DEFAULT 0,
  PRIMARY KEY (id_producto)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ==========================================
-- 3. TABLA HIJA (con FK + CASCADE)
-- ==========================================

CREATE TABLE pedidos (
  id_pedido    INT        NOT NULL AUTO_INCREMENT,
  id_cliente   INT        NOT NULL,
  id_producto  INT        NULL,
  cantidad     INT        NOT NULL DEFAULT 1,
  fecha_pedido DATETIME   NOT NULL DEFAULT NOW(),
  estado       ENUM('pendiente','enviado','entregado','cancelado')
               NOT NULL DEFAULT 'pendiente',
  PRIMARY KEY (id_pedido),
  CONSTRAINT fk_pedido_cliente
    FOREIGN KEY (id_cliente)
    REFERENCES clientes(id_cliente)
    ON DELETE CASCADE
    ON UPDATE CASCADE,
  CONSTRAINT fk_pedido_producto
    FOREIGN KEY (id_producto)
    REFERENCES productos(id_producto)
    ON DELETE SET NULL
    ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- ==========================================
-- 4. ÍNDICES ADICIONALES
-- ==========================================

CREATE INDEX idx_cliente_nombre ON clientes(nombre);
CREATE INDEX idx_pedido_estado  ON pedidos(estado);

-- ==========================================
-- 5. DATOS DE PRUEBA
-- ==========================================

INSERT INTO clientes (nombre, correo, telefono) VALUES
  ('Ana Gómez',   'ana@correo.com',   '3001234567'),
  ('Luis Torres', 'luis@correo.com',  '3109876543'),
  ('María Ríos',  'maria@correo.com', NULL);

INSERT INTO productos (nombre, precio, stock) VALUES
  ('Cuaderno',  4500.00,  50),
  ('Lapicero',  1200.00,  200),
  ('Mochila',   85000.00, 15);

INSERT INTO pedidos (id_cliente, id_producto, cantidad, estado) VALUES
  (1, 2, 3, 'pendiente'),
  (2, 1, 1, 'enviado'),
  (1, 3, 2, 'pendiente');

-- ==========================================
-- 6. VERIFICACIÓN FINAL
-- ==========================================

SELECT * FROM clientes;
SELECT * FROM productos;
SELECT
  p.id_pedido,
  c.nombre   AS cliente,
  pr.nombre  AS producto,
  p.cantidad,
  p.estado
FROM pedidos p
JOIN clientes  c  ON p.id_cliente  = c.id_cliente
LEFT JOIN productos pr ON p.id_producto = pr.id_producto
ORDER BY p.id_pedido;
🚀
Este script usa DROP DATABASE IF EXISTS al inicio, lo que permite ejecutarlo múltiples veces sin errores. Ideal para pruebas y reinicios durante práctica en clase.
