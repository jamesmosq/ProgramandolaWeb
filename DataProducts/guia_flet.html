01
Instalación y requisitos
Python 3.10+ · pip · venv · flet doctor · Windows/Mac/Linux
▼

Flet requiere Python 3.10 o superior y funciona en Windows 10/11, macOS 12+ y Linux (Debian/Ubuntu). Se recomienda siempre usar un entorno virtual.

💡
Flet es esencialmente Flutter para Python. El motor gráfico es Flutter internamente, pero tú escribes solo Python. Esto significa que el mismo código corre como app nativa, web y móvil.

Instalar con pip (recomendado para Windows)

Terminal
# Crear carpeta del proyecto
mkdir mi-app-flet
cd mi-app-flet

# Crear entorno virtual
python -m venv .venv
.venv\Scripts\activate       # Windows
# source .venv/bin/activate  # Mac/Linux

# Instalar Flet con todos los extras (desktop + web + mobile)
pip install 'flet[all]'

# Verificar instalación
flet --version
flet doctor              # diagnóstico completo del entorno

Instalar con uv (alternativa más rápida)

Terminal
# uv es un gestor de paquetes ultra-rápido escrito en Rust
pip install uv

uv init --python='>=3.10'
uv venv
uv add 'flet[all]'

# Verificar
uv run flet --version
uv run flet doctor

Actualizar Flet

Terminal
pip install 'flet[all]' --upgrade
02
Crear y correr la primera app
flet create · ft.run · page · hot reload · --web · --android
▼

Toda app Flet tiene una función main(page) que recibe el objeto page. Ahí construyes la UI agregando controles con page.add().

Crear proyecto con flet create

Terminal
# Crea la estructura del proyecto
flet create

# Estructura generada:
# README.md
# pyproject.toml
# src/
#   assets/
#     icon.png
#   main.py
#   storage/data/ storage/temp/

Hola Mundo — src/main.py

Python
import flet as ft

def main(page: ft.Page):
    page.title = "Mi primera app Flet"
    page.add(ft.Text("¡Hola, mundo!"))

ft.run(main)

Modos de ejecución

Terminal
# App de escritorio (ventana nativa)
flet run src/main.py

# App web (abre en el navegador)
flet run --web src/main.py

# Con hot reload — recarga automática al guardar
flet run -r src/main.py

# Puerto personalizado para web
flet run --web --port 8080 src/main.py
✅
Auto-update: en Flet moderno NO necesitas llamar page.update() manualmente. Se llama automáticamente al final de cada event handler. Solo desactívalo si haces 100+ cambios en un loop (por rendimiento).

App contador — ejemplo completo

Python — Contador
import flet as ft

def main(page: ft.Page):
    page.title = "Contador"
    page.vertical_alignment = ft.MainAxisAlignment.CENTER

    # Control TextField para mostrar el valor
    contador = ft.TextField(
        value="0",
        text_align=ft.TextAlign.RIGHT,
        width=100
    )

    def restar(e):
        contador.value = str(int(contador.value) - 1)
        # page.update() se llama automáticamente

    def sumar(e):
        contador.value = str(int(contador.value) + 1)

    page.add(
        ft.Row(
            alignment=ft.MainAxisAlignment.CENTER,
            controls=[
                ft.IconButton(ft.Icons.REMOVE, on_click=restar),
                contador,
                ft.IconButton(ft.Icons.ADD, on_click=sumar),
            ]
        )
    )

ft.run(main)

import flet as ft

def main(page: ft.Page):
    page.title = "Demo controles"
    page.padding = 20

    # Texto con estilo
    titulo = ft.Text(
        "Registro de Estudiante",
        size=24,
        weight=ft.FontWeight.BOLD,
        color=ft.Colors.BLUE_700
    )

    # Campo de texto
    nombre = ft.TextField(
        label="Nombre completo",
        hint_text="Ej: Ana Gómez",
        prefix_icon=ft.Icons.PERSON
    )

    correo = ft.TextField(
        label="Correo",
        keyboard_type=ft.KeyboardType.EMAIL,
        prefix_icon=ft.Icons.EMAIL
    )

    # Dropdown
    programa = ft.Dropdown(
        label="Programa",
        options=[
            ft.dropdown.Option("Análisis de Sistemas"),
            ft.dropdown.Option("Desarrollo de Software"),
            ft.dropdown.Option("Bases de Datos"),
        ]
    )

    # Checkbox y Switch
    activo = ft.Checkbox(label="Estudiante activo", value=True)
    notif  = ft.Switch(label="Recibir notificaciones", value=False)

    # Resultado
    resultado = ft.Text("", color=ft.Colors.GREEN_700)

    def guardar(e):
        resultado.value = (
            f"Guardado: {nombre.value} | {programa.value}"
        )

    btn = ft.ElevatedButton(
        "Guardar",
        icon=ft.Icons.SAVE,
        on_click=guardar
    )

    page.add(titulo, nombre, correo, programa, activo, notif, btn, resultado)

ft.run(main)

Layout — Row, Column, Container, expand
Row · Column · Container · expand · Stack · SafeArea · alignment
▼

Flet usa el mismo sistema de layout que Flutter: Row (horizontal), Column (vertical) y Container (decoración). La propiedad expand distribuye el espacio disponible.

Python — Layout completo
import flet as ft

def main(page: ft.Page):

    # Row — elementos horizontales
    fila = ft.Row(
        controls=[
            ft.Text("Izquierda"),
            ft.Text("Centro"),
            ft.Text("Derecha"),
        ],
        alignment=ft.MainAxisAlignment.SPACE_BETWEEN,
        vertical_alignment=ft.CrossAxisAlignment.CENTER,
    )

    # Column — elementos verticales
    columna = ft.Column(
        controls=[
            ft.Text("Primero"),
            ft.Text("Segundo"),
            ft.Text("Tercero"),
        ],
        spacing=10,
        scroll=ft.ScrollMode.AUTO,  # columna scrolleable
    )

    # Container — agrupa y decora un control
    caja = ft.Container(
        content=ft.Text("Contenido", color=ft.Colors.WHITE),
        bgcolor=ft.Colors.BLUE_700,
        border_radius=ft.BorderRadius.all(12),
        padding=20,
        width=200,
        alignment=ft.alignment.center,
    )

    # expand — distribuir espacio proporcional
    barra = ft.Row(
        controls=[
            ft.Container(expand=1, bgcolor=ft.Colors.RED_300, height=50),
            ft.Container(expand=3, bgcolor=ft.Colors.BLUE_300, height=50),
            ft.Container(expand=1, bgcolor=ft.Colors.GREEN_300, height=50),
        ],
        spacing=8,
    )
    # Resultado: 20% | 60% | 20%  (1+3+1 = 5 partes)

    # Layout de pantalla completa con sidebar + contenido
    layout = ft.Row(
        controls=[
            # Sidebar fijo
            ft.Container(
                width=200,
                bgcolor=ft.Colors.BLUE_GREY_900,
                content=ft.Column(controls=[
                    ft.Text("Menú", color=ft.Colors.WHITE),
                ])
            ),
            # Contenido principal expandible
            ft.Container(
                expand=True,
                content=ft.Text("Área principal"),
                padding=20,
            )
        ],
        expand=True,
    )

    page.add(fila, columna, caja, barra)

ft.run(main)
💡
expand=True ocupa todo el espacio disponible. expand=N divide el espacio proporcionalmente. En el ejemplo 1+3+1: los extremos tienen 20% y el centro 60%.

Eventos e interactividad
on_click · on_change · on_submit · page.dialog · SnackBar · AlertDialog
▼
Python — Eventos y diálogos
import flet as ft

def main(page: ft.Page):
    page.title = "Eventos"

    campo = ft.TextField(label="Escribe algo")
    output = ft.Text("")

    # on_change — se dispara con cada tecla
    def al_cambiar(e):
        output.value = f"Escribiendo: {campo.value}"

    # on_submit — se dispara al presionar Enter
    def al_enviar(e):
        output.value = f"Enviado: {campo.value}"
        campo.value = ""

    campo.on_change = al_cambiar
    campo.on_submit = al_enviar

    # SnackBar — notificación temporal
    def mostrar_snack(e):
        page.open(ft.SnackBar(
            content=ft.Text("✅ Guardado correctamente"),
            bgcolor=ft.Colors.GREEN_700,
        ))

    # AlertDialog — ventana de confirmación
    def mostrar_dialogo(e):
        def confirmar(e):
            page.close(dlg)
            output.value = "Confirmado"

        dlg = ft.AlertDialog(
            modal=True,
            title=ft.Text("Confirmar"),
            content=ft.Text("¿Estás seguro?"),
            actions=[
                ft.TextButton("Cancelar", on_click=lambda e: page.close(dlg)),
                ft.ElevatedButton("Aceptar", on_click=confirmar),
            ]
        )
        page.open(dlg)

    # BottomSheet
    def mostrar_sheet(e):
        bs = ft.BottomSheet(
            content=ft.Container(
                padding=20,
                content=ft.Column(
                    tight=True,
                    controls=[
                        ft.Text("Opciones", size=18, weight=ft.FontWeight.BOLD),
                        ft.ElevatedButton("Opción 1"),
                        ft.ElevatedButton("Opción 2"),
                    ]
                )
            )
        )
        page.open(bs)

    page.add(
        campo,
        output,
        ft.Row([
            ft.ElevatedButton("Snack", on_click=mostrar_snack),
            ft.ElevatedButton("Diálogo", on_click=mostrar_dialogo),
            ft.ElevatedButton("Sheet", on_click=mostrar_sheet),
        ])
    )

ft.run(main)

Controles personalizados (Custom Controls)
@ft.control · herencia · @dataclass · props tipadas · reusable
▼

El decorador @ft.control permite crear componentes reutilizables con propiedades tipadas. Equivale a los widgets de Flutter o los componentes de Vue/React.

Python — Controles custom
import flet as ft

# Control base con propiedades comunes
@ft.control
class CalcButton(ft.Button):
    expand: int = 1

# Herencia — botón para dígitos
@ft.control
class DigitButton(CalcButton):
    bgcolor: ft.Colors = ft.Colors.WHITE_24
    color:   ft.Colors = ft.Colors.WHITE

# Botón de operación (naranja)
@ft.control
class ActionButton(CalcButton):
    bgcolor: ft.Colors = ft.Colors.ORANGE
    color:   ft.Colors = ft.Colors.WHITE

# Control complejo — tarjeta de estudiante
@ft.control
class TarjetaEstudiante(ft.Container):
    nombre:   str  = ""
    ficha:    str  = ""
    activo:   bool = True

    def build(self):
        return ft.Container(
            bgcolor=ft.Colors.BLUE_GREY_900,
            border_radius=12,
            padding=16,
            content=ft.Row(controls=[
                ft.Icon(ft.Icons.PERSON, color=ft.Colors.WHITE),
                ft.Column(controls=[
                    ft.Text(self.nombre, color=ft.Colors.WHITE, weight=ft.FontWeight.BOLD),
                    ft.Text(f"Ficha: {self.ficha}", color=ft.Colors.BLUE_200, size=12),
                ], expand=True),
                ft.Icon(
                    ft.Icons.CHECK_CIRCLE if self.activo else ft.Icons.CANCEL,
                    color=ft.Colors.GREEN_400 if self.activo else ft.Colors.RED_400
                )
            ])
        )

def main(page: ft.Page):
    page.add(
        TarjetaEstudiante(nombre="Ana Gómez",    ficha="2876543", activo=True),
        TarjetaEstudiante(nombre="Luis Torres",  ficha="2876544", activo=False),
    )

ft.run(main)

Python — Navegación con múltiples vistas
import flet as ft

def main(page: ft.Page):
    page.title = "EduCode"

    # AppBar — barra superior
    page.appbar = ft.AppBar(
        leading=ft.Icon(ft.Icons.SCHOOL),
        title=ft.Text("EduCode"),
        bgcolor=ft.Colors.BLUE_700,
        color=ft.Colors.WHITE,
        actions=[
            ft.IconButton(ft.Icons.SETTINGS, icon_color=ft.Colors.WHITE),
        ]
    )

    # Contenido de cada sección
    def vista_cursos():
        return ft.Column([
            ft.Text("Cursos disponibles", size=20, weight=ft.FontWeight.BOLD),
            ft.Text("Bases de Datos"),
            ft.Text("PHP Puro"),
            ft.Text("Laravel 13"),
        ], spacing=10)

    def vista_perfil():
        return ft.Column([
            ft.Text("Mi perfil", size=20, weight=ft.FontWeight.BOLD),
            ft.Text("James Mosquera"),
            ft.Text("Instructor EduCode"),
        ])

    contenido = ft.Column([vista_cursos()])

    # NavigationBar — barra inferior
    def cambiar_tab(e):
        idx = e.control.selected_index
        contenido.controls.clear()
        if idx == 0:
            contenido.controls.append(vista_cursos())
        elif idx == 1:
            contenido.controls.append(vista_perfil())

    page.navigation_bar = ft.NavigationBar(
        destinations=[
            ft.NavigationBarDestination(icon=ft.Icons.BOOK, label="Cursos"),
            ft.NavigationBarDestination(icon=ft.Icons.PERSON, label="Perfil"),
        ],
        on_change=cambiar_tab,
    )

    page.add(contenido)

ft.run(main)

Temas, colores y tipografía
page.theme · ThemeMode · ft.Colors · ColorScheme · dark mode
▼
Python — Temas completos
import flet as ft

def main(page: ft.Page):
    page.title = "EduCode"

    # Tema personalizado
    page.theme = ft.Theme(
        color_scheme_seed=ft.Colors.BLUE,   # genera paleta desde este color
        use_material3=True,
        font_family="Roboto",
    )

    # Modo oscuro / claro / sistema
    page.theme_mode = ft.ThemeMode.SYSTEM   # LIGHT, DARK, SYSTEM

    def toggle_theme(e):
        page.theme_mode = (
            ft.ThemeMode.DARK
            if page.theme_mode == ft.ThemeMode.LIGHT
            else ft.ThemeMode.LIGHT
        )

    # Colores predefinidos
    # ft.Colors.BLUE_700, GREEN_400, RED_300, ORANGE, etc.
    # ft.Colors.WHITE, BLACK, TRANSPARENT

    # Texto con estilo completo
    texto = ft.Text(
        "EduCode",
        size=32,
        weight=ft.FontWeight.W_900,
        color=ft.Colors.BLUE_700,
        italic=False,
        font_family="Roboto",
        text_align=ft.TextAlign.CENTER,
    )

    page.add(
        texto,
        ft.ElevatedButton(
            "🌙 Cambiar tema",
            on_click=toggle_theme
        )
    )

ft.run(main)
