01
Instalación de Moonshine 4.x
composer · moonshine:install · moonshine_users · /admin
▼

Moonshine 4.x es el panel de administración para Laravel 13. Genera un sistema completo con su propio sistema de auth, UI reactiva y componentes CRUD listos en minutos.

🌙
Moonshine 4.x funciona sobre Laravel 13 + PHP 8.3. Tiene su propia tabla moonshine_users separada de la tabla users de la app. El admin del panel y los usuarios de la app son independientes.

Instalar el paquete

Terminal
# 1. Instalar el paquete
composer require moonshine/moonshine

# 2. Ejecutar el instalador interactivo
php artisan moonshine:install

# El instalador te preguntará:
# ✓ ¿Instalar migraciones de Moonshine? → yes
# ✓ ¿Crear usuario administrador? → yes
# ✓ Nombre del admin → Tu nombre
# ✓ Email del admin → tu@correo.com
# ✓ Password → ********

# 3. Ejecutar migraciones
php artisan migrate

# 4. Compilar assets de Moonshine
php artisan moonshine:publish

# 5. Acceder al panel
# http://tu-dominio.test/admin

Configuración del panel
config/moonshine.php · título · colores · prefix · locale
▼
PHP — config/moonshine.php
return [
    // Título que aparece en la barra lateral y el browser tab
    'title'  => 'EduCode Admin',

    // Logo — archivo en public/
    'logo'   => '/img/logo.svg',

    // Prefijo de la URL del panel
    'prefix' => 'admin',    // → /admin

    // Idioma del panel
    'locale' => 'es',

    // Middleware aplicado a todas las rutas del panel
    'middleware' => [
        'web',
        \MoonShine\Http\Middleware\Authenticate::class,
    ],

    // Colores del panel (usa hex o tailwind)
    'colors' => [
        'primary'   => '#3b82f6',   // azul principal
        'secondary' => '#8b5cf6',   // violeta
    ],

    // Usuario administrador de Moonshine
    'auth' => [
        'guard'      => 'moonshine',
        'model'      => \MoonShine\Models\MoonshineUser::class,
        'enable'     => true,
    ],
];

03
Primer ModelResource — CRUD completo
moonshine:resource · ModelResource · fields() · column
▼

Un ModelResource en Moonshine 4.x genera automáticamente listado, creación, edición y eliminación para un modelo Eloquent. Es el equivalente a un CRUD completo en un solo archivo.

Terminal
# Crear resource para el modelo Curso
php artisan moonshine:resource CursoResource --model=Curso

# Ver todos los comandos de Moonshine
php artisan moonshine:list
PHP — app/MoonShine/Resources/CursoResource.php
<?php
namespace App\MoonShine\Resources;

use App\Models\Curso;
use MoonShine\Laravel\Resources\ModelResource;
use MoonShine\UI\Fields\ID;
use MoonShine\UI\Fields\Text;
use MoonShine\UI\Fields\Number;
use MoonShine\UI\Fields\Select;
use MoonShine\UI\Fields\Switcher;
use MoonShine\UI\Fields\Textarea;

class CursoResource extends ModelResource
{
    // Modelo Eloquent asociado
    protected string $model = Curso::class;

    // Título en el panel
    protected string $title = 'Cursos';

    // Columna que se usa para representar el registro
    // (en dropdowns de relaciones)
    protected string $column = 'nombre';

    // Campos del listado Y del formulario
    protected function indexFields(): iterable
    {
        return [
            ID::make()->sortable(),
            Text::make('Nombre', 'nombre')->sortable(),
            Text::make('Instructor', 'instructor'),
            Number::make('Horas', 'horas')->sortable(),
            Select::make('Nivel', 'nivel')
                ->options([
                    'basico'      => 'Básico',
                    'intermedio'  => 'Intermedio',
                    'avanzado'    => 'Avanzado',
                ]),
            Switcher::make('Activo', 'activo'),
        ];
    }

    // Campos del formulario (crear/editar)
    protected function formFields(): iterable
    {
        return [
            ID::make(),
            Text::make('Nombre', 'nombre')
                ->required()
                ->placeholder('Ej: Bases de Datos'),
            Text::make('Instructor', 'instructor')
                ->required(),
            Number::make('Horas', 'horas')
                ->min(1)->default(40),
            Select::make('Nivel', 'nivel')
                ->options([
                    'basico'     => 'Básico',
                    'intermedio' => 'Intermedio',
                    'avanzado'   => 'Avanzado',
                ])->required(),
            Textarea::make('Descripción', 'descripcion')
                ->nullable(),
            Switcher::make('Activo', 'activo')
                ->default(true),
        ];
    }
}
🌙
Moonshine 4.x separó indexFields() (campos en la tabla listado) de formFields() (campos en el formulario). Esto da control total sobre qué se muestra en cada vista.

Ejemplo: Fields con métodos avanzados

PHP — Fields con modificadores
use MoonShine\UI\Fields\Image;
use MoonShine\UI\Fields\TinyMce;
use MoonShine\UI\Fields\Slug;
use MoonShine\UI\Fields\Select;

// Imagen con disco y carpeta
Image::make('Portada', 'imagen')
    ->disk('public')
    ->dir('cursos')
    ->nullable()
    ->hideOnIndex(),

// Editor de contenido enriquecido
TinyMce::make('Contenido', 'contenido')
    ->hideOnIndex(),

// Slug que se genera desde el nombre
Slug::make('URL', 'slug')
    ->from('nombre')  // genera slug desde campo 'nombre'
    ->unique(),

// Select con badge de colores
Select::make('Estado', 'estado')
    ->options([
        'pendiente'  => 'Pendiente',
        'activo'     => 'Activo',
        'cancelado'  => 'Cancelado',
    ])
    ->badge(fn($val) => match($val) {
        'pendiente' => 'warning',
        'activo'    => 'success',
        'cancelado' => 'danger',
        default     => 'default',
    }),

// Mostrar / ocultar según contexto
Text::make('Nombre', 'nombre')
    ->showOnIndex()      // visible en el listado
    ->showOnForm()       // visible en crear/editar
    ->hideOnDetail()     // oculto en la vista detalle
    ->sortable()
    ->searchable(),
05
Menú lateral — registrar Resources
MoonShineServiceProvider · MenuItem · MenuGroup · icons
▼

El menú del panel se define en el MoonShineServiceProvider. Puedes crear grupos, separadores y asignar íconos de Heroicons.

PHP — app/MoonShine/MoonShineServiceProvider.php
<?php
namespace App\MoonShine;

use MoonShine\Laravel\DependencyInjection\MoonShineConfigurator;
use MoonShine\Laravel\DependencyInjection\ConfiguratorContract;
use MoonShine\MenuManager\MenuItem;
use MoonShine\MenuManager\MenuGroup;
use MoonShine\MenuManager\MenuDivider;

// Importar todos los resources
use App\MoonShine\Resources\CursoResource;
use App\MoonShine\Resources\EstudianteResource;
use App\MoonShine\Resources\InstructorResource;
use App\MoonShine\Resources\MatriculaResource;
use App\MoonShine\Resources\NotaResource;

class MoonShineServiceProvider
{
    public function __invoke(
        ConfiguratorContract $config
    ): ConfiguratorContract {

        return $config
            ->menu([

                MenuItem::make(
                    'Dashboard',
                    static fn() => new \App\MoonShine\Pages\Dashboard(),
                    'heroicons.home'
                ),

                MenuDivider::make(),

                MenuGroup::make('Académico', [
                    MenuItem::make(
                        'Cursos',
                        static fn() => new CursoResource(),
                        'heroicons.academic-cap'
                    ),
                    MenuItem::make(
                        'Instructores',
                        static fn() => new InstructorResource(),
                        'heroicons.user'
                    ),
                ]),

                MenuGroup::make('Estudiantes', [
                    MenuItem::make(
                        'Estudiantes',
                        static fn() => new EstudianteResource(),
                        'heroicons.users'
                    ),
                    MenuItem::make(
                        'Matrículas',
                        static fn() => new MatriculaResource(),
                        'heroicons.clipboard-document-list'
                    ),
                    MenuItem::make(
                        'Notas',
                        static fn() => new NotaResource(),
                        'heroicons.star'
                    ),
                ]),

            ]);
    }
}

Filtros y búsqueda
filters() · search · $search · RangeSlider · DateRange
▼
PHP — Búsqueda y Filtros en Resource
use MoonShine\UI\Fields\RangeSlider;
use MoonShine\UI\Fields\DateRange;

class CursoResource extends ModelResource
{
    // Columnas buscables (barra de búsqueda del listado)
    protected array $search = ['nombre', 'instructor', 'descripcion'];

    // Panel de filtros laterales
    protected function filters(): iterable
    {
        return [
            Text::make('Instructor', 'instructor')
                ->like('instructor'),   // LIKE %valor%

            Select::make('Nivel', 'nivel')
                ->options([
                    'basico'     => 'Básico',
                    'intermedio' => 'Intermedio',
                    'avanzado'   => 'Avanzado',
                ])->nullable(),

            RangeSlider::make('Horas', 'horas')
                ->fromTo('horas_min', 'horas_max')
                ->min(0)->max(200),

            DateRange::make('Creado', 'created_at')
                ->fromTo('desde', 'hasta'),

            Switcher::make('Solo activos', 'activo'),
        ];
    }

    // Ordenamiento por defecto
    protected function modifyQueryBuilder(\Illuminate\Database\Eloquent\Builder $query): \Illuminate\Database\Eloquent\Builder
    {
        return $query->orderBy('nombre', 'asc');
    }
}
07
Relaciones en Moonshine 4.x
BelongsTo · HasMany · BelongsToMany · RelationRepeater
▼
PHP — Relaciones en formFields()
use MoonShine\UI\Fields\Relationships\BelongsTo;
use MoonShine\UI\Fields\Relationships\HasMany;
use MoonShine\UI\Fields\Relationships\BelongsToMany;
use MoonShine\UI\Fields\Relationships\RelationRepeater;

// BelongsTo — Matrícula pertenece a un Estudiante
BelongsTo::make(
    'Estudiante',
    'estudiante',
    static fn() => new EstudianteResource()
)
->required()
->searchable(),   // búsqueda AJAX en dropdown

// BelongsTo — Matrícula pertenece a un Curso
BelongsTo::make(
    'Curso',
    'curso',
    static fn() => new CursoResource()
)
->required()
->searchable(),

// HasMany — Curso tiene muchas matrículas
// (se muestra en la página de detalle del Curso)
HasMany::make(
    'Matrículas',
    'matriculas',
    static fn() => new MatriculaResource()
)
->hideOnIndex()   // solo en la vista detalle
->hideOnForm(),

// BelongsToMany — Estudiante ↔ Cursos (tabla pivot)
BelongsToMany::make(
    'Cursos',
    'cursos',
    static fn() => new CursoResource()
)
->searchable(),
✅
->searchable() en un BelongsTo activa búsqueda por AJAX — indispensable cuando tienes cientos de registros. Sin él, carga todos los registros en el dropdown.

Proyecto real — Sistema académico con 5 módulos
Cursos · Instructores · Estudiantes · Matrículas · Notas
▼

Construimos un sistema académico completo desde cero. 5 módulos con sus relaciones, resources y menú organizado.

Los 5 módulos del proyecto

📚
Cursos
nombre, horas, nivel, instructor_id, activo
👨‍🏫
Instructores
nombre, especialidad, correo, activo
👨‍🎓
Estudiantes
nombre, documento, ficha, programa, activo
📋
Matrículas
estudiante_id, curso_id, fecha, estado
⭐
Notas
matricula_id, nota, periodo, observacion

Paso 1 — Crear modelos y migraciones

Terminal
# Crear todos los modelos con migración
php artisan make:model Instructor -m
php artisan make:model Curso -m
php artisan make:model Estudiante -m
php artisan make:model Matricula -m
php artisan make:model Nota -m

Paso 2 — Migraciones en orden correcto (padre → hijo)

PHP — Migraciones de los 5 módulos
// 1. instructores (sin FK)
$table->id();
$table->string('nombre', 100);
$table->string('especialidad', 100);
$table->string('correo')->unique();
$table->boolean('activo')->default(true);
$table->timestamps();

// 2. cursos (FK → instructores)
$table->id();
$table->string('nombre', 100);
$table->text('descripcion')->nullable();
$table->unsignedInteger('horas')->default(40);
$table->enum('nivel', ['basico','intermedio','avanzado'])->default('basico');
$table->foreignId('instructor_id')->constrained('instructores')->restrictOnDelete();
$table->boolean('activo')->default(true);
$table->timestamps();

// 3. estudiantes (sin FK)
$table->id();
$table->string('nombre', 150);
$table->string('documento', 20)->unique();
$table->string('ficha', 20)->nullable();
$table->string('programa', 100)->nullable();
$table->boolean('activo')->default(true);
$table->timestamps();

// 4. matriculas (FK → estudiantes + cursos)
$table->id();
$table->foreignId('estudiante_id')->constrained('estudiantes')->cascadeOnDelete();
$table->foreignId('curso_id')->constrained('cursos')->restrictOnDelete();
$table->date('fecha_matricula');
$table->enum('estado',['activa','retirada','finalizada'])->default('activa');
$table->timestamps();

// 5. notas (FK → matriculas)
$table->id();
$table->foreignId('matricula_id')->constrained('matriculas')->cascadeOnDelete();
$table->decimal('nota', 4, 1);
$table->string('periodo', 20); // Ej: "2024-1"
$table->text('observacion')->nullable();
$table->timestamps();

Paso 4 — NotaResource con relaciones completas

PHP — NotaResource.php
class NotaResource extends ModelResource
{
    protected string $model = Nota::class;
    protected string $title = 'Notas';
    protected array  $search = ['matricula.estudiante.nombre', 'periodo'];

    protected function indexFields(): iterable
    {
        return [
            ID::make()->sortable(),
            BelongsTo::make(
                'Matrícula', 'matricula',
                static fn() => new MatriculaResource()
            ),
            Number::make('Nota', 'nota')->sortable(),
            Text::make('Período', 'periodo'),
        ];
    }

    protected function formFields(): iterable
    {
        return [
            ID::make(),
            BelongsTo::make(
                'Matrícula', 'matricula',
                static fn() => new MatriculaResource()
            )->required()->searchable(),
            Number::make('Nota', 'nota')
                ->required()->min(0)->max(5)->step(0.1),
            Text::make('Período', 'periodo')
                ->required()
                ->placeholder('Ej: 2024-1'),
            Textarea::make('Observación', 'observacion')->nullable(),
        ];
    }
}

Deploy del panel en Railway
railway up · moonshine:user · storage:link · assets
▼
Terminal — Checklist de deploy
# 1. Subir el código a Railway
railway up

# 2. Migrar en producción
railway run php artisan migrate --force

# 3. Crear usuario admin de Moonshine en producción
railway run php artisan moonshine:user

# 4. Publicar assets de Moonshine
railway run php artisan moonshine:publish

# 5. Vincular storage para imágenes subidas
railway run php artisan storage:link

# 6. Cachear para producción
railway run php artisan config:cache
railway run php artisan route:cache
railway run php artisan view:cache
