Instalación del SDK de Flutter
flutter SDK · Dart · Android Studio · VS Code · flutter doctor
▼

Flutter incluye Dart integrado. Solo necesitas descargar el SDK, configurar el PATH y verificar con flutter doctor.

🐦
Flutter compila a código nativo ARM/x86 para móvil, bytecode JS para web, y ejecutables nativos para desktop. No usa WebView ni puentes JS — es código nativo real en todas las plataformas.

Instalación en Windows

Terminal — Windows
# Opción A: con winget (recomendado)
winget install Google.Flutter

# Opción B: descargar SDK manualmente
# docs.flutter.dev/install/quick → descargar zip
# Extraer en C:\flutter
# Agregar C:\flutter\bin al PATH del sistema

# Verificar instalación completa
flutter doctor

# Acepta licencias de Android
flutter doctor --android-licenses

Instalación en macOS / Linux

Terminal — Mac/Linux
# macOS con Homebrew
brew install flutter

# Linux — snap
sudo snap install flutter --classic

# Verificar
flutter doctor -v

# Crear primer proyecto
flutter create mi_app
cd mi_app
flutter run

variables · tipos · funciones · clases · null safety · async/await

Flutter usa Dart, un lenguaje tipado, compilado, con null safety desde Dart 2.12. Si ya sabes Java o C# te va a resultar familiar.

Estado con StatefulWidget
setState() · State · initState · dispose · TextEditingController
▼
Dart — StatefulWidget completo
import 'package:flutter/material.dart';

// Widget con estado
class ContadorScreen extends StatefulWidget {
  const ContadorScreen({super.key});

  @override
  State<ContadorScreen> createState() => _ContadorScreenState();
}

class _ContadorScreenState extends State<ContadorScreen> {
  // Variables de estado
  int _contador = 0;
  final TextEditingController _ctrl = TextEditingController();

  // Ciclo de vida: se llama una vez al crear el widget
  @override
  void initState() {
    super.initState();
    _ctrl.text = '0';
  }

  // Ciclo de vida: limpiar recursos al destruir
  @override
  void dispose() {
    _ctrl.dispose();
    super.dispose();
  }

  void _incrementar() {
    setState(() {        // setState dispara un rebuild del widget
      _contador++;
    });
  }

  void _decrementar() {
    setState(() {
      if (_contador > 0) _contador--;
    });
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('Contador')),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            Text(
              '$_contador',
              style: Theme.of(context).textTheme.displayLarge,
            ),
            const SizedBox(height: 24),
            Row(
              mainAxisAlignment: MainAxisAlignment.center,
              children: [
                IconButton(
                  icon: const Icon(Icons.remove),
                  onPressed: _decrementar,
                  iconSize: 36,
                ),
                const SizedBox(width: 24),
                IconButton(
                  icon: const Icon(Icons.add),
                  onPressed: _incrementar,
                  iconSize: 36,
                ),
              ],
            ),
          ],
        ),
      ),
    );
  }
}

Dart — Formulario con validación
class FormularioScreen extends StatefulWidget {
  const FormularioScreen({super.key});
  @override
  State<FormularioScreen> createState() => _FormularioScreenState();
}

class _FormularioScreenState extends State<FormularioScreen> {
  // Key global para acceder al estado del Form
  final _formKey = GlobalKey<FormState>();

  // Controladores para leer valores
  final _nombreCtrl = TextEditingController();
  final _correoCtrl = TextEditingController();
  String? _nivelSeleccionado;

  @override
  void dispose() {
    _nombreCtrl.dispose();
    _correoCtrl.dispose();
    super.dispose();
  }

  void _guardar() {
    if (_formKey.currentState!.validate()) {
      // Todos los campos son válidos
      ScaffoldMessenger.of(context).showSnackBar(
        const SnackBar(content: Text('✅ Guardado correctamente')),
      );
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('Registro')),
      body: Padding(
        padding: const EdgeInsets.all(16),
        child: Form(
          key: _formKey,
          autovalidateMode: AutovalidateMode.onUserInteraction,
          child: Column(children: [

            // Campo nombre
            TextFormField(
              controller: _nombreCtrl,
              decoration: const InputDecoration(
                labelText: 'Nombre *',
                prefixIcon: Icon(Icons.person),
                border: OutlineInputBorder(),
              ),
              validator: (val) {
                if (val == null || val.isEmpty) {
                  return 'El nombre es obligatorio';
                }
                if (val.length < 3) {
                  return 'Mínimo 3 caracteres';
                }
                return null; // null = válido
              },
            ),
            const SizedBox(height: 16),

            // Campo correo
            TextFormField(
              controller: _correoCtrl,
              keyboardType: TextInputType.emailAddress,
              decoration: const InputDecoration(
                labelText: 'Correo *',
                prefixIcon: Icon(Icons.email),
                border: OutlineInputBorder(),
              ),
              validator: (val) {
                if (val == null || !val.contains('@')) {
                  return 'Ingresa un correo válido';
                }
                return null;
              },
            ),
            const SizedBox(height: 16),

            // Dropdown
            DropdownButtonFormField<String>(
              value: _nivelSeleccionado,
              decoration: const InputDecoration(
                labelText: 'Nivel',
                border: OutlineInputBorder(),
              ),
              items: ['Básico', 'Intermedio', 'Avanzado']
                  .map((n) => DropdownMenuItem(value: n, child: Text(n)))
                  .toList(),
              onChanged: (val) => setState(() => _nivelSeleccionado = val),
              validator: (val) => val == null ? 'Selecciona un nivel' : null,
            ),
            const SizedBox(height: 24),

            ElevatedButton.icon(
              onPressed: _guardar,
              icon: const Icon(Icons.save),
              label: const Text('Guardar'),
              style: ElevatedButton.styleFrom(minimumSize: const Size.fromHeight(48)),
            ),
          ]),
        ),
      ),
    );
  }
}

class Curso {
  final String nombre;
  final String instructor;
  final int    horas;
  final String nivel;
  bool         activo;

  Curso({
    required this.nombre,
    required this.instructor,
    required this.horas,
    required this.nivel,
    this.activo = true,
  });

  // Datos de prueba
  static List<Curso> ejemplos = [
    Curso(nombre: 'Bases de Datos',  instructor: 'James', horas: 40, nivel: 'Básico'),
    Curso(nombre: 'PHP Puro',         instructor: 'James', horas: 36, nivel: 'Básico'),
    Curso(nombre: 'Laravel 13',       instructor: 'James', horas: 48, nivel: 'Intermedio'),
    Curso(nombre: 'Flutter',           instructor: 'James', horas: 60, nivel: 'Avanzado'),
    Curso(nombre: 'Moonshine 4.x',    instructor: 'James', horas: 32, nivel: 'Intermedio'),
  ];
}

screens/cursos_screen.dart

Dart — Lista de cursos
class CursosScreen extends StatefulWidget {
  const CursosScreen({super.key});
  @override
  State<CursosScreen> createState() => _CursosScreenState();
}

class _CursosScreenState extends State<CursosScreen> {
  final cursos = Curso.ejemplos;

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: const Text('Cursos EduCode')),
      body: ListView.builder(
        padding: const EdgeInsets.all(12),
        itemCount: cursos.length,
        itemBuilder: (ctx, i) {
          final c = cursos[i];
          return Card(
            margin: const EdgeInsets.only(bottom: 10),
            child: ListTile(
              leading: CircleAvatar(
                backgroundColor: Colors.blue,
                child: Text('${i+1}', style: const TextStyle(color: Colors.white)),
              ),
              title: Text(c.nombre, style: const TextStyle(fontWeight: FontWeight.bold)),
              subtitle: Text('${c.instructor} · ${c.horas}h · ${c.nivel}'),
              trailing: Switch(
                value: c.activo,
                onChanged: (v) => setState(() => c.activo = v),
              ),
              onTap: () => Navigator.push(
                context,
                MaterialPageRoute(builder: (_) => DetalleScreen(curso: c)),
              ),
            ),
          );
        },
      ),
      floatingActionButton: FloatingActionButton.extended(
        onPressed: () => Navigator.push(
          context,
          MaterialPageRoute(builder: (_) => const FormularioScreen()),
        ),
        icon: const Icon(Icons.add),
        label: const Text('Nuevo curso'),
      ),
    );
  }
}
